#include<iostream>
#include<algorithm>
using namespace std;
int a[1005];
int main()
{
	int n;
	scanf("%d",&n);
	int i;
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	int mid=(1+n)/2,c1=0,c2=0;
	for(i=1;i<mid;i++)
	{
		if(a[i]<a[mid])
			c1++;
		else
			break;
	}
	for(i=n;i>mid;i--)
	{
		if(a[i]>a[mid])
			c2++;
		else
			break;
	}
	if(c1==c2)
		printf("%d\n",a[mid]);
	else
		printf("-1\n");
	return 0;
}
